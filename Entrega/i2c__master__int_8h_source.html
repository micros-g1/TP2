5  * @brief I2C Master Interface
6  * @details This interface is prepared to handle communications using any I2C module as a master in a non-blocking way.

36 /**
37  * @brief I2C Master Read Data
38  * @details sends a read data package from a specific I2C module.
39  * If specified, the read package will be preceded by a write package with information (for the slave)
40  * about the type of reading that will be performed. If this package is not needed, then it will not be sent.
41  * @param mod_id : I2C module from which data will be read.
42  * @param question : reading information transferred (with a write frame) to the slave before the reading sequence
43  * (Repeated Start + Address + Read) is sent.
44  * @param len_question : amount of bytes question has. If 0, then no write package will be sent.
45  * @param amount_of_bytes : amount of bytes to be received when the reading is performed.
46  */
47 void i2c_master_int_read_data(i2c_module_id_int_t mod_id, unsigned char* question, int len_question, int amount_of_bytes);
48 
49 /**
50  * @brief I2C Master has new data
51  * @details ask if a specific module has new data in the buffer that has not been read yet.
52  * @param mod : I2C module to be asked if new data has been received.
53  * @return *true* if new data has arrived to the buffer. *false* otherwise
54  */
55 bool i2c_master_int_has_new_data(i2c_module_id_int_t mod_id);
56 
57 /**
58  * @brief I2C Master get new data length
59  * @details ask for the length of the new data a specific module has in the buffer and that has not been read yet.
60  * The user has to know that new data has arrived. See also i2c_master_int_has_new_data() .
61  * @param mod_id : I2C module to be asked the new data's length.
62  * @return amount of bytes the new data has.
63  */
64 int i2c_master_int_get_new_data_length(i2c_module_id_int_t mod_id);
65 
66 /**
67  * @brief I2C Master get new data
68  * @details gets new data from the I2C buffer, following FIFO convention.
69  * Data obtained by the use of this function is deleted from the buffer.
70  * @param mod_id : I2C module to be asked for the new data.
71  * @param read_data : user's buffer into which new data should be loaded.
72  * @param amount_of_bytes : amount of bytes to read from the I2C specific module's buffer.
73  */
74 void i2c_master_int_get_new_data(i2c_module_id_int_t mod_id, unsigned char* read_data, int amount_of_bytes);
75 
76 /**
77  * @brief I2C Master write data
78  * @details Send a Write frame with a specific module.
79  * The slave address the write frames will refer to should be set before calling this function.
80  * For this, see also i2c_master_int_set_slave_addr() .
81  * @param mod_id : I2C module that will write the data to the slave.
82  * @param write_data : data that will be sent to the slave.
83  * @param amount_of_bytes : amount of bytes that will be sent to the slave (length of write_data parameter).
84  */
85 void i2c_master_int_write_data(i2c_module_id_int_t mod_id, unsigned char* write_data, int amount_of_bytes);
86 
87 /**
88  * @brief I2C Master set slave address
89  * @details set the slave adress a specific I2C module will refer its frames to.
90  * @param mod_id : I2C module for which the slave's address will be set.
91  */
92 void i2c_master_int_set_slave_addr(i2c_module_id_int_t mod_id, unsigned char slave_add);
93

95 /**
96  * @brief I2C Master BUS is busy
97  * @details Get the current status of the I2C BUS for a specific I2C module ( Busy or not)
98  * The Bus is busy when every START signal sent to the bus has encountered its corresponding STOP signal in the bus
99  * (both sent by this master module).
100  * @param mod_id : I2C module for which the bus status will be checked.
101  * @return *true* is the bus is currently busy. *false* otherwise.
102  */
103 bool i2c_master_int_bus_busy(i2c_module_id_int_t mod_id);
